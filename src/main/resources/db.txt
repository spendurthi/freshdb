create table DEPT(
  SID INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 100, INCREMENT BY 1) PRIMARY KEY, 
  DEPT_ID INTEGER NOT NULL UNIQUE,
  DEPT_NAME varchar(15) NOT NULL,
  DEPT_LOC varchar(20)  
);

create table EMP(
  SID INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 100, INCREMENT BY 1) PRIMARY KEY,
  ID INTEGER NOT NULL UNIQUE,
  NAME varchar(20) NOT NULL UNIQUE,
  JOB      varchar(10) NOT NULL UNIQUE,
  MGR      INTEGER,
  HIREDATE date,
  SAL      INTEGER,
  COMM     INTEGER,
  DEPT_ID   INTEGER,
  constraint fk_deptno foreign key (DEPT_ID) references DEPT(DEPT_ID)
);

CREATE TABLE user (
  SID INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 100, INCREMENT BY 1) PRIMARY KEY,
  USER_NAME VARCHAR(12) not null,
  PASSWORD VARCHAR(12) not null,
  FIRST_NAME VARCHAR(15) not null,
  LAST_NAME VARCHAR(15) not null,
  MIDDLE_NAME VARCHAR(15)
);
